Constructing Analytics Barplot

means_data <- data.frame(
  Characteristic = c("Danceability", "Loudness", "Energy", "Speechiness", "Acousticness", "Instrumentalness"),
  Mean = c(dance, loud, energy, speechiness, acoustic, instrumental)
)

  theme_minimal() +
